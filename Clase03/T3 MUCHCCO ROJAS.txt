
SELECT CUR.CUR_ID, CUR.CUR_NOMBRE, CUR.CUR_MATRICULADOS, 
(CUR.CUR_MATRICULADOS * CUR.CUR_PRECIO) IMPORTEPROYECTADO, 
NVL(IMPORTERECAUDADO,0), 
NVL(IMPORTECOMPROMETIDO,0)
FROM EDUCA.CURSO CUR
FULL JOIN (
	SELECT CUR_ID,SUM(PAG_IMPORTE) IMPORTERECAUDADO FROM EDUCA.PAGO GROUP BY CUR_ID
) PA ON CUR.CUR_ID=PA.CUR_ID
FULL JOIN (
	SELECT CUR_ID,SUM(MAT_PRECIO) IMPORTECOMPROMETIDO FROM EDUCA.MATRICULA GROUP BY CUR_ID
)MA ON CUR.CUR_ID=MA.CUR_ID;
